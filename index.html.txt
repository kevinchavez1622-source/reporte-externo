function margenFor(q){ for(const[a,b,m]of MARGEN) if(q>=a&&q<=b) return m; return 0.50; }
function rangoLbl(q){ for(const[a,b,m]of MARGEN) if(q>=a&&q<=b) return b===Infinity?`${a.toLocaleString()}+`:`${a.toLocaleString()}–${b.toLocaleString()}`; return ""; }
const mx = n => "$"+n.toLocaleString("es-MX",{minimumFractionDigits:2,maximumFractionDigits:2});

function compute(){
  const q = Math.max(200, state.qty||200);
  const precioUnit = PRECIOS[state.dim][state.mat];
  const costoBase = precioUnit*q;
  const margen = margenFor(q);
  const subtotal = costoBase*(1+margen);
  const envio = state.ship==="guia" ? GUIA : 0;
  const conEnvio = subtotal+envio;
  const iva = conEnvio*IVA;
  const total = conEnvio+iva;
  return {q,precioUnit,costoBase,margen,subtotal,envio,iva,total,porPieza:total/q};
}

async function vrun(logId, steps){
  const log = document.getElementById(logId); log.innerHTML="";
  for(const s of steps){
    const row=document.createElement("div"); row.className="vl wait";
    row.innerHTML=`<span class="vi">⟳</span><span>${s.t}</span>`;
    log.appendChild(row); await sleep(s.ms);
    row.className="vl "+(s.fail?"fail":"pass"); row.querySelector(".vi").textContent = s.fail?"✕":"✓";
  }
}

function vbubble(chat, side, html, tick){
  const d=document.createElement("div"); d.className="vmsg "+side;
  const t=new Date(); const hh=t.getHours().toString().padStart(2,"0")+":"+t.getMinutes().toString().padStart(2,"0");
  d.innerHTML = html + `<span class="vt">${hh}${tick?'<span class="tk"> ✓✓</span>':''}</span>`;
  chat.appendChild(d); return d;
}
